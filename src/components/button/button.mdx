## You can also have a more complex example

<Playground>
  {() => {
    class ExampleButton extends React.Component {
      constructor(props) {
        super(props);
        this.state = { inverted: false };
        this.handleClick = this.handleClick.bind(this);
      }

      handleClick(e) {
        e.preventDefault();
        this.setState(prevState => ({ inverted: !prevState.inverted }));
      }

      render() {
        const { inverted } = this.state;
        return (
          <Button onClick={this.handleClick} inverted={inverted}>
            {inverted ? "Inverted" : "Default"}
          </Button>
        );
      }
    }
    return <ExampleButton />;
  }
}
</Playground>
